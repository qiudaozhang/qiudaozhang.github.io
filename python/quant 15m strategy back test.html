class Break_It(BaseTest):

    def __init__(self, data, balance, margin, leverage, commission, fix_win, fix_loss):
        super().__init__(data, balance, margin, leverage, commission)
        self.fix_win = fix_win
        self.fix_loss = fix_loss

    def close_try(self):
        if self.has_position:
            if self.max_loss() < self.fix_loss:
                self.close_profit(self.fix_loss)
                print("k inner loss")

    def next(self):
        super().next()
        if self.has_position:
            self.out()
        else:
            self.entry()

    def entry(self):
        df = self.df[:-1]
        last = df.iloc[-1]
        k = self.now_k
        if last['ma'] > last['ma2'] > last['ma3']:
            self.buy_long(k['o'])
            return
        if last['ma'] < last['ma2'] < last['ma3']:
            self.buy_short(k['o'])
            return

    def out(self):
        df = self.df[:-1]
        last = df.iloc[-1]
        k = self.now_k
        if self.is_long():
            if last['ma'] < last['ma2'] < last['ma3']:
                self.close(k['o'])
                return

        if self.is_short():
            if last['ma'] > last['ma2'] > last['ma3']:
                self.close(k['o'])
                return
